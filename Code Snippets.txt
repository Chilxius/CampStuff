  void bounce( Chaser c )
  {
    //Directional Bounce
    if( xPos > c.xPos )
      xSpd = abs(c.xSpd);
    else
      xSpd = -abs(c.xSpd);
      
    if( yPos > c.yPos )
      ySpd = abs(c.ySpd);
    else
      ySpd = -abs(c.ySpd);
      
    bounceDelay = 10;
  }
  
  boolean touching( Chaser c )
  {
    return( dist( c.xPos, c.yPos, xPos, yPos ) < (c.size+size)/2 );
  }
